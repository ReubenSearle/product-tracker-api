query newlyListedUsedProductsInLocalArea {
	productSearch(
    input: {
      filter: {
        buyerPostCode: "CT19 5HE"
        category: garageEquipmentTools
        condition: used
        maxDistance: 5
      }
      sort: startTimeNewest
    }
  ) {
  	products {
      ...productResult
    }
	}
}

fragment productResult on Product {
  location {
    name
  }
  productId
  productTitle
}
